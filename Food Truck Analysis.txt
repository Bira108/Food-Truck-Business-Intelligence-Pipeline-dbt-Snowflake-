stg_orders.sql:


{{config(materialized = 'table')}}

WITH max_date_cte AS (
    SELECT MAX(ORDER_TS) AS max_date
    FROM TASTY_BYTES_SAMPLE_DATA.RAW_POS.ORDER_HEADER
)
SELECT
    o.ORDER_ID AS order_id,
    -- If CUSTOMER_ID is null, replace it with 0 so the test passes safely
    COALESCE(o.CUSTOMER_ID, 0) AS customer_id,
    o.ORDER_TOTAL AS total_price,   
    o.ORDER_TS AS order_date
FROM TASTY_BYTES_SAMPLE_DATA.RAW_POS.ORDER_HEADER o
JOIN max_date_cte m
    ON o.ORDER_TS >= DATEADD('day', -30, m.max_date)


stg_lineitems.sql:


{{ config(materialized = 'table')}}

WITH max_date_cte AS (
    SELECT MAX(ORDER_TS) AS max_date
    FROM TASTY_BYTES_SAMPLE_DATA.RAW_POS.ORDER_HEADER
)
SELECT
    d.ORDER_DETAIL_ID AS order_detail_id,
    d.ORDER_ID AS order_id,
    d.MENU_ITEM_ID AS menu_item_id,
    d.QUANTITY AS quantity,
    d.UNIT_PRICE AS unit_price,
    d.PRICE AS line_price,
    d.ORDER_ITEM_DISCOUNT_AMOUNT AS discount_amount,
    h.ORDER_TS AS order_ts
FROM TASTY_BYTES_SAMPLE_DATA.RAW_POS.ORDER_DETAIL d
JOIN TASTY_BYTES_SAMPLE_DATA.RAW_POS.ORDER_HEADER h
    ON d.ORDER_ID = h.ORDER_ID
JOIN max_date_cte m
    ON h.ORDER_TS >= DATEADD('day', -30, m.max_date)


mart_customer_orders.sql:


{{ config(materialized='table') }}

SELECT
    o.order_id,
    o.customer_id,
    o.total_price,
    l.quantity,
    l.menu_item_id
FROM {{ ref('stg_orders') }} AS o
JOIN {{ ref('stg_lineitems') }} AS l
    ON o.order_id = l.order_id
    


mart_monthly_orders.sql


{{ config(materialized='table') }}

SELECT
    DATE_TRUNC('month', o.order_date) AS order_month,
    o.customer_id,
    SUM(o.total_price) AS total_price,
    SUM(l.quantity) AS total_quantity,
    COUNT(o.order_id) AS total_orders
FROM
    {{ ref('stg_orders') }} AS o
JOIN
    {{ ref('stg_lineitems') }} AS l
    ON o.order_id = l.order_id
GROUP BY
    1, 2


recent_orders:

-- incremental models --

{{ config(materialized='incremental', unique_key='order_date') }}
WITH recent_orders AS (
    SELECT
        DATE_TRUNC('day', DATEADD(year, 31, o.order_date)) AS order_date,
        SUM(o.total_price) AS total_revenue,
        COUNT(o.order_id) AS total_orders
    FROM {{ ref('stg_orders') }} AS o
    WHERE o.order_date <= CURRENT_DATE - INTERVAL '30 days'
    GROUP BY 1
)

SELECT
    order_date,
    total_revenue,
    total_orders
FROM recent_orders
{% if is_incremental() %}
  WHERE order_date > (SELECT MAX(order_date) FROM {{ this }})
{% endif %}
ORDER BY order_date
